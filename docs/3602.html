有些标题有一个值数组，定义它是否有一个下拉列表。

在将JSON传递给Exceljs服务之前，让我们创建一个函数来转换它

transform(data) {
    const noOfRowaToGenerate = 10;
    return data.map(({
        name,
        values
    }) => {
        const headers = values.reduce(
            (prev, next) => ({
                ...prev,
                [next.header]: Array.isArray(next.value) ?
                    next.value.map(({
                        name
                    }) => name) :
                    next.value
            }), {}
        );
        return {
            workSheet: name,
            rows: Array(noOfRowaToGenerate).fill(headers)
        };
    });
}
